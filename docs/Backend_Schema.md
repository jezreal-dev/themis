## Version 1.0 | AMD AI DevMaster Hackathon | July 20, 2026

## 1. Overview

This document defines all data schemas used in the THEMIS backend:
- Pydantic models for API request/response validation
- LangGraph state schema (ThemisState)
- Qdrant vector collection configurations
- WebSocket event schemas
- Internal agent data models
- Benchmark data schema

All schemas use **Pydantic v2** syntax.

## 2. Core Domain Models

### 2.1 Finding

The atomic unit of output from the analysis pipeline.

```python
# backend/models/finding.py
from pydantic import BaseModel, Field, field_validator
from typing import Optional, Literal
from uuid import UUID, uuid4
import json
import os

VALID_CWE_IDS: set[str] = set(
    json.load(open(os.path.join(os.path.dirname(__file__), "../../data/cwe_id_lookup.json"))).keys()
)

class RAGCitation(BaseModel):
    source: str                         # "CWE-89" | "OWASP A03:2021" | "PEP8 E501"
    excerpt: str                        # First 300 chars of retrieved chunk
    url: Optional[str] = None           # Link to authoritative source

class Finding(BaseModel):
    id: UUID = Field(default_factory=uuid4)
    type: Literal["security", "style"]
    severity: Literal["critical", "high", "medium", "low", "info"]

    # Location
    file: str                           # Relative file path: "src/auth.py"
    line_start: int = Field(ge=1)
    line_end: int = Field(ge=1)

    # Content
    title: str = Field(min_length=5, max_length=200)
    description: str = Field(min_length=10)

    # Authority
    cwe_id: Optional[str] = None        # "CWE-89"
    cwe_name: Optional[str] = None      # "SQL Injection"
    cwe_valid: bool = False             # Set by cwe_validator.py
    owasp_ref: Optional[str] = None     # "A03:2021-Injection"
    rag_citations: list[RAGCitation] = []

    # Provenance
    tool_source: Literal["semgrep", "bandit", "pylint", "eslint", "llm"]
    raw_tool_output: Optional[str] = None

    # Verification (set by Verifier Agent)
    confidence: float = Field(ge=0.0, le=1.0, default=0.0)
    verified: bool = False              # True if confidence >= 0.7

    # Code evidence
    original_snippet: Optional[str] = None   # Vulnerable code
    suggested_snippet: Optional[str] = None  # Fixed code (if LLM-generated)

    @field_validator("cwe_id")
    @classmethod
    def validate_cwe_id(cls, v: Optional[str]) -> Optional[str]:
        if v is not None and v not in VALID_CWE_IDS:
            # Don't reject — let cwe_valid flag communicate invalidity
            return v
        return v

    @field_validator("line_end")
    @classmethod
    def end_after_start(cls, v: int, info) -> int:
        if "line_start" in info.data and v < info.data["line_start"]:
            raise ValueError("line_end must be >= line_start")
        return v
```

## 3. LangGraph State Schema

```python
# backend/agents/state.py
from typing import TypedDict, Annotated, Optional
import operator
from uuid import UUID
from .models.finding import Finding
from .models.patch import Patch
from .models.triage import TriagedFile

class ThemisState(TypedDict):

    # ─── Job Identity ──────────────────────────────────────────────────
    job_id: str                           # UUID string
    created_at: str                       # ISO 8601 timestamp

    # ─── Input ─────────────────────────────────────────────────────────
    input_type: str                       # "github_pr" | "file_upload"

    # GitHub PR inputs
    repo: Optional[str]                   # "owner/repo"
    pr_number: Optional[int]
    pr_title: Optional[str]
    pr_base_sha: Optional[str]
    pr_head_sha: Optional[str]
    pr_author: Optional[str]

    # Raw diff (sanitized)
    raw_diff: str                         # Sanitized by sanitizer.py
    diff_retrieval_method: str            # "api" | "git_clone_fallback"

    # ─── Triage Agent Output ───────────────────────────────────────────
    triage_complete: bool
    prioritized_files: list[TriagedFile]  # Ordered by risk_score DESC
    language_map: dict[str, str]          # {filepath: language}
    total_files: int
    total_additions: int
    total_deletions: int

    # ─── Parallel Agent Outputs (LangGraph reducers = list merge) ──────
    security_findings: Annotated[list[Finding], operator.add]
    style_findings: Annotated[list[Finding], operator.add]
    agent_errors: Annotated[list[str], operator.add]

    # ─── Verifier Agent Output ─────────────────────────────────────────
    verifier_complete: bool
    verified_findings: list[Finding]          # confidence >= 0.7
    low_confidence_findings: list[Finding]    # confidence < 0.7

    # ─── Human Gate ────────────────────────────────────────────────────
    awaiting_approval: bool
    fix_approved: bool

    # ─── Fix Agent Output ──────────────────────────────────────────────
    fix_complete: bool
    patches: list[Patch]
    fix_branch: Optional[str]               # "themis/fix-pr-42"
    fix_pr_url: Optional[str]               # Full GitHub PR URL
    fix_pr_number: Optional[int]

    # ─── Safety / Loop Guards ──────────────────────────────────────────
    visited_tool_hashes: set[str]           # SHA-256 of (tool_name, args) pairs
    step_count: int                         # Increments each agent step

    # ─── Metrics ───────────────────────────────────────────────────────
    total_tokens_input: int
    total_tokens_output: int
    elapsed_seconds: Optional[float]
    pipeline_status: str                    # "running" | "awaiting_approval" | "complete" | "error"
```

All WebSocket messages are JSON. Client receives only; no client → server messages after connection.

```python
# backend/models/events.py
from pydantic import BaseModel
from typing import Any, Literal, Optional

AgentName = Literal["triage", "security", "style", "verifier", "fix"]

    "connected",
    "agent_started",
    "tool_call",

    "finding_added",

    event: EventType
```

## 7. Job Storage Schema

Jobs are stored **in-memory** (dict) during the hackathon. No database required.

```python
# backend/state/job_store.py

from typing import dict
import asyncio

class JobStore:
    """Thread-safe in-memory job store"""

    def __init__(self):
        self._jobs: dict[str, dict] = {}
        self._lock = asyncio.Lock()

    async def create(self, job_id: str, initial_state: dict) -> None:
        async with self._lock:
            self._jobs[job_id] = {
                "state": initial_state,
                "ws_queue": asyncio.Queue(),  # Events queued for WebSocket
                "created_at": datetime.utcnow().isoformat()
            }

    async def get(self, job_id: str) -> Optional[dict]:
        return self._jobs.get(job_id)

    async def update_state(self, job_id: str, updates: dict) -> None:
        async with self._lock:
            if job_id in self._jobs:
                self._jobs[job_id]["state"].update(updates)

    async def enqueue_event(self, job_id: str, event: dict) -> None:
        if job_id in self._jobs:
            await self._jobs[job_id]["ws_queue"].put(event)

# Singleton
job_store = JobStore()
```
